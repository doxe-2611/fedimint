16/// Test note consolidation, which at the time of writing basically means that
17/// once client accumulates too many notes of certain denomination, any
18/// transaction building will include excessive notes as extra inputs, to
19/// consolidate them into higher denominations.
20///
21/// In the future we will probably change the whole thing and delete this thing.
22async fn test_note_consoliation(fed: &devimint::federation::Federation) -> anyhow::Result<()> {
23    let sender = fed.new_joined_client("sender").await?;
24    let receiver = fed.new_joined_client("receiver").await?;
25
26    let can_no_wait = cmd!(sender, "reissue", "--help")
27        .out_string()
28        .await?
29        .contains("no-wait");
30
31    if !can_no_wait {
32        info!("Version before `--no-wait` didn't have consolidation implemented");
33        return Ok(());
34    }
35    fed.pegin_client(10_000, &sender).await?;
36
37    let mut all_notes = vec![];
38    for i in 0..20 {
39        let info = cmd!(sender, "info").out_json().await?;
40        info!(%info, "sender info");
41        // remint sender notes from time to time to make sure it have 1msat notes
42        if i % 2 == 1 {
43            let notes = cmd!(sender, "spend", "1sat",).out_json().await?["notes"]
44                .as_str()
45                .context("invoice must be string")?
46                .to_owned();
47
48            cmd!(sender, "reissue", notes).run().await?;
49        }
50
51        let notes = cmd!(sender, "spend", "1msat",).out_json().await?["notes"]
52            .as_str()
53            .context("invoice must be string")?
54            .to_owned();
55
56        all_notes.push(notes);
57    }
58
59    for notes in &all_notes[..all_notes.len() - 1] {
60        cmd!(receiver, "reissue", "--no-wait", notes).run().await?;
61    }
62
63    // wait for all at the same time to make things go faster
64    cmd!(receiver, "dev", "wait-complete").run().await?;
65
66    // reissuance of last note will trigger consolidation
67    cmd!(receiver, "reissue")
68        .args(&all_notes[all_notes.len() - 1..])
69        .run()
70        .await?;
71
72    let info = cmd!(receiver, "info").out_json().await?;
73    info!(%info, "receiver info");
74    // receiver has the balance
75    assert_eq!(info["total_amount_msat"].as_i64().unwrap(), 20);
76    // without the consolidation, this would be 20 1msat notes
77    assert!(info["denominations_msat"]["1"].as_i64().unwrap() < 20);
78
79    Ok(())
80}
